5   package org.epics.util.text;
6   
7   import java.text.NumberFormat;
8   import static org.hamcrest.CoreMatchers.equalTo;
9   import org.junit.Test;
10  import static org.junit.Assert.*;
11  import static org.hamcrest.Matchers.*;
12  
13  /**
14   *
15   * @author carcassi
16   */
17  public class NumberFormatsTest {
18      
19      public NumberFormatsTest() {
20      }
21  
22      @Test
23      public void format1() {
24          NumberFormat format = NumberFormats.format(2);
25          assertThat(format.format(2.0), equalTo("2.00"));
26          assertThat(format.format(Double.NaN), equalTo("NaN"));
27          assertThat(format.format(Double.POSITIVE_INFINITY), equalTo("Infinity"));
28          assertThat(format.format(Double.NEGATIVE_INFINITY), equalTo("-Infinity"));
29          assertThat(NumberFormats.format(2), sameInstance(format));
30      }
31      
32      @Test
33      public void format2() {
34          NumberFormat f = NumberFormats.format(3);
35          assertThat(f.format(1234.4567), equalTo("1234.457"));
36          assertThat(f.format(123), equalTo("123.000"));
37          assertThat(f.format(123.4), equalTo("123.400"));
38  
39          f = NumberFormats.format(0);
40          assertThat(f.format(1234.4567), equalTo("1234"));
41          assertThat(f.format(123), equalTo("123"));
42          assertThat(f.format(123.4), equalTo("123"));
43  
44          f = NumberFormats.format(4);
45          assertThat(f.format(1234.4567), equalTo("1234.4567"));
46          assertThat(f.format(123), equalTo("123.0000"));
47          assertThat(f.format(123.4), equalTo("123.4000"));
48      }
49  
50      @Test
51      public void toStringFormat() {
52          NumberFormat format = NumberFormats.toStringFormat();
53          assertThat(format.format(2.0), equalTo("2.0"));
54          assertThat(format.format(Double.NaN), equalTo("NaN"));
55          assertThat(format.format(Double.POSITIVE_INFINITY), equalTo("Infinity"));
56          assertThat(format.format(Double.NEGATIVE_INFINITY), equalTo("-Infinity"));
57          assertThat(NumberFormats.toStringFormat(), sameInstance(format));
58      }
59  }
