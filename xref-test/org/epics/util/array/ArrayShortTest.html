7   import org.junit.Test;
8   import static org.junit.Assert.*;
9   import static org.hamcrest.Matchers.*;
10  
11  /**
12   *
13   * @author carcassi

15  public class ArrayShortTest {
16      
17      public ArrayShortTest() {
18      }
19  
20      @Test
21      public void wrap1() {
22          ArrayShort array = new ArrayShort(new short[] {0, 1, 2, 3, 4, 5});
23          assertThat(CollectionNumbers.doubleArrayCopyOf(array), equalTo(new double[] {0, 1, 2, 3, 4, 5}));
24      }
25  
26      @Test(expected=UnsupportedOperationException.class)
27      public void wrap2() {
28          ArrayShort array = new ArrayShort(new short[] {0, 1, 2, 3, 4, 5});
29          array.setDouble(0, 0);
30      }
31  
32      @Test
33      public void wrap3() {
34          ArrayShort array = new ArrayShort(new short[] {0, 1, 2, 3, 4, 5}, false);
35          array.setDouble(0, 5);
36          array.setDouble(5, 0);
37          assertThat(CollectionNumbers.doubleArrayCopyOf(array), equalTo(new double[] {5, 1, 2, 3, 4, 0}));
38      }
39  }
