18      @Test
19      public void sortedView1() {
20          ArrayDouble values = new ArrayDouble(5,3,1,4,2,0);
21          SortedListView sortedView = ListNumbers.sortedView(values);
22          assertThat(values, equalTo(new ArrayDouble(5,3,1,4,2,0)));
23          assertThat(sortedView, equalTo((ListNumber) new ArrayDouble(0,1,2,3,4,5)));
24          assertThat(sortedView.getIndexes(), equalTo((ListInt) new ArrayInt(5,2,4,1,3,0)));
25      }

27      @Test
28      public void sortedView2() {
29          ArrayDouble values = new ArrayDouble(5,3,1,4,2,0);
30          ArrayInt indexes = new ArrayInt(0,3,1,4,2,5);
31          SortedListView sortedView = ListNumbers.sortedView(values, indexes);
32          assertThat(values, equalTo(new ArrayDouble(5,3,1,4,2,0)));

210         assertThat(isLinear(linearListFromRange(0, 100, 10000)), equalTo(true));
211         assertThat(isLinear(ListMath.add(linearList(0, 0.00001, 10000), 3.0)), equalTo(true));
212         assertThat(isLinear(linearListFromRange(0, 100, 10000)), equalTo(true));
213         assertThat(isLinear(new ArrayDouble(0,1,2,3,4,5)), equalTo(true));
214         assertThat(isLinear(new ArrayDouble(0,1.00001,2,3,4,5)), equalTo(false));
