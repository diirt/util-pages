5   package org.epics.util.stats;
6   
7   import java.util.ArrayList;
8   import java.util.Arrays;
9   import java.util.List;
10  import org.epics.util.array.ArrayDouble;
11  import org.junit.Test;
12  import static org.junit.Assert.*;
13  import static org.hamcrest.CoreMatchers.*;
14  
15  
16  /**
17   *
18   * @author carcassi
19   */
20  public class StatisticsUtilTest {
21      
22      @Test
23      public void statisticsOf1() {
24          Statistics stats = StatisticsUtil.statisticsOf(new ArrayDouble(1.0));
25          assertThat(stats.getAverage(), equalTo(1.0));
26          assertThat(stats.getStdDev(), equalTo(0.0));
27          assertThat(stats.getMinimum(), equalTo((Number) 1.0));
28          assertThat(stats.getMaximum(), equalTo((Number) 1.0));
29          assertThat(stats.getCount(), equalTo(1));
30      }
31      
32      @Test
33      public void statisticsOf2() {
34          Statistics stats = StatisticsUtil.statisticsOf(new ArrayDouble(1, 3, 5, -1, 7));
35          assertThat(stats.getAverage(), equalTo(3.0));
36          assertThat(stats.getStdDev(), equalTo(2.8284271247461903));
37          assertThat(stats.getMinimum(), equalTo((Number) (-1.0)));
38          assertThat(stats.getMaximum(), equalTo((Number) 7.0));
39          assertThat(stats.getCount(), equalTo(5));
40      }
41      
42      @Test
43      public void statisticsOf3() {
44          List<Statistics> list = new ArrayList<Statistics>();
45          for (int i = 0; i < 10; i++) {
46              list.add(StatisticsUtil.statisticsOf(new ArrayDouble(i)));
47          }
48          Statistics stats = StatisticsUtil.statisticsOf(list);
49          assertThat(stats.getAverage(), equalTo(4.5));
50          assertThat(stats.getStdDev(), equalTo(2.8722813232690143));
51          assertThat(stats.getMinimum(), equalTo((Number) 0.0));
52          assertThat(stats.getMaximum(), equalTo((Number) 9.0));
53          assertThat(stats.getCount(), equalTo(10));
54      }
55      
56      @Test
57      public void statisticsOf4() {
58          Statistics stats = StatisticsUtil.statisticsOf(new ArrayDouble(1, 3, 5, Double.NaN, -1, 7));
59          assertThat(stats.getAverage(), equalTo(3.0));
60          assertThat(stats.getStdDev(), equalTo(2.8284271247461903));
61          assertThat(stats.getMinimum(), equalTo((Number) (-1.0)));
62          assertThat(stats.getMaximum(), equalTo((Number) 7.0));
63          assertThat(stats.getCount(), equalTo(5));
64      }
65      
66      @Test
67      public void statisticsOf5() {
68          Statistics stats = StatisticsUtil.statisticsOf(new ArrayDouble(1, 3, 5, -1, 7));
69          stats = StatisticsUtil.statisticsOf(Arrays.asList(stats));
70          assertThat(stats.getAverage(), equalTo(3.0));
71          assertThat(stats.getStdDev(), equalTo(2.8284271247461903));
72          assertThat(stats.getMinimum(), equalTo((Number) (-1.0)));
73          assertThat(stats.getMaximum(), equalTo((Number) 7.0));
74          assertThat(stats.getCount(), equalTo(5));
75      }
76  }
