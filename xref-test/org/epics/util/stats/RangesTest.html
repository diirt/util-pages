38      @Test
39      public void subrange1() {
40          assertThat(Ranges.contains(Ranges.range(0.0, 1.0), Ranges.range(0.5, 0.75)), equalTo(true));
41          assertThat(Ranges.contains(Ranges.range(0.0, 1.0), Ranges.range(0.5, 1.0)), equalTo(true));
42          assertThat(Ranges.contains(Ranges.range(0.0, 1.0), Ranges.range(0.0, 0.75)), equalTo(true));
43          assertThat(Ranges.contains(Ranges.range(0.0, 1.0), Ranges.range(-1.0, 0.75)), equalTo(false));
44          assertThat(Ranges.contains(Ranges.range(0.0, 1.0), Ranges.range(0.0, 1.75)), equalTo(false));
45      }

47      @Test
48      public void sum1() {
49          Range range1 = Ranges.range(0.0, 5.0);
50          Range range2 = Ranges.range(1.0, 2.0);
51          assertThat(Ranges.sum(range1, range2), sameInstance(range1));
52          assertThat(Ranges.sum(range2, range1), sameInstance(range1));
53      }

55      @Test
56      public void sum2() {
57          Range range1 = Ranges.range(0.0, 5.0);
58          Range range2 = Ranges.range(1.0, 6.0);
59          Range range = Ranges.sum(range1, range2);
60          assertThat(range.getMinimum(), equalTo((Number) 0.0));
61          assertThat(range.getMaximum(), equalTo((Number) 6.0));
62          range = Ranges.sum(range2, range1);
63          assertThat(range.getMinimum(), equalTo((Number) 0.0));
64          assertThat(range.getMaximum(), equalTo((Number) 6.0));
