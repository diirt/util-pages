7   import java.text.ParseException;

17  public class TimestampFormatTest {
18  
19      public TimestampFormatTest() {
20      }
21  
22      @Test
23      public void formatTimestamp1() {
24          // Test with milliseconds
25          TimestampFormat format = new TimestampFormat("yyyy-MM-dd'T'HH:mm:ss.S");
26          format.setTimeZome(TimeZone.getTimeZone("GMT"));
27          Timestamp time = Timestamp.of(0, 30000000);
28          assertThat(format.format(time), equalTo("1970-01-01T00:00:00.30"));
29      }
30  
31      @Test
32      public void formatTimestamp2() {
33          // Test with nanoseconds
34          TimestampFormat format = new TimestampFormat("yyyy-MM-dd'T'HH:mm:ss.N");
35          format.setTimeZome(TimeZone.getTimeZone("GMT"));
36          Timestamp time = Timestamp.of(0, 30000000);
37          assertThat(format.format(time), equalTo("1970-01-01T00:00:00.30000000"));
38      }
39  
40      @Test
41      public void spacedFormat() {
42          TimestampFormat format = new TimestampFormat("yyyy-MM-dd'T'HH:mm:ss.NNNNNN");
43          format.setTimeZome(TimeZone.getTimeZone("GMT"));
44          Timestamp time = Timestamp.of(0, 1);
45          assertThat(format.format(time), equalTo("1970-01-01T00:00:00.000001"));
46          time = Timestamp.of(0, 12);
47          assertThat(format.format(time), equalTo("1970-01-01T00:00:00.000012"));
48          time = Timestamp.of(0, 123);
49          assertThat(format.format(time), equalTo("1970-01-01T00:00:00.000123"));
50          time = Timestamp.of(0, 1234);
51          assertThat(format.format(time), equalTo("1970-01-01T00:00:00.001234"));
52          time = Timestamp.of(0, 12345);
53          assertThat(format.format(time), equalTo("1970-01-01T00:00:00.012345"));
54          time = Timestamp.of(0, 123456);
55          assertThat(format.format(time), equalTo("1970-01-01T00:00:00.123456"));
56          time = Timestamp.of(0, 1234567);
57          assertThat(format.format(time), equalTo("1970-01-01T00:00:00.1234567"));
58          time = Timestamp.of(0, 12345678);
59          assertThat(format.format(time), equalTo("1970-01-01T00:00:00.12345678"));
60      }
61  
62      @Test
63      public void specialCases() {
64          // Only nanoseconds
65          TimestampFormat format = new TimestampFormat("NNN");
66          format.setTimeZome(TimeZone.getTimeZone("GMT"));
67          Timestamp time = Timestamp.of(0, 12345);
68          assertThat(format.format(time), equalTo("12345"));
69  
70          // Multiple nanoseconds with different format
71          // and N as part of escaped text
72          format = new TimestampFormat("NNNNN-'''N'''-N");
73          format.setTimeZome(TimeZone.getTimeZone("GMT"));
74          time = Timestamp.of(0, 1);
75          assertThat(format.format(time), equalTo("00001-'N'-1"));
76      }
77      
78      @Test
79      public void parse1() throws Exception {
80          TimestampFormat format = new TimestampFormat("yyyy-MM-dd'T'HH:mm:ss");
81          format.setTimeZome(TimeZone.getTimeZone("GMT"));
82          Timestamp time = format.parse("1976-01-01T00:00:00");
83          assertThat(time, equalTo(Timestamp.of(189302400, 0)));
84      }
85      
86      @Test(expected=ParseException.class)
87      public void parse2() throws Exception {
88          TimestampFormat format = new TimestampFormat("yyyy-MM-dd'T'HH:mm:ss");
89          format.setTimeZome(TimeZone.getTimeZone("GMT"));
90          Timestamp time = format.parse("1976-NN-01T00:00:00");
91      }
