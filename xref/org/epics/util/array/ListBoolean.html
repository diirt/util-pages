5   package org.epics.util.array;
6   
7   /**
8    * An ordered collection of {@code boolean}s. Since in Java Boolean does
9    * not inherit from Number, ListBoolean does not inherit from ListNumber.
10   *
11   * @author Gabriele Carcassi
12   */
13  public abstract class ListBoolean {
14      
15      /**
16       * Returns the element at the specified position in this list.
17       *
18       * @param index position of the element to return
19       * @return the element at the specified position in this list
20       * @throws IndexOutOfBoundsException if the index is out of range
21       *         (<tt>index &lt; 0 || index &gt;= size()</tt>)
22       */
23      public abstract boolean getBoolean(int index);
24  
25      /**
26       * Changes the element at the specified position.
27       * 
28       * @param index position of the element to change
29       * @param value the new value
30       * @throws IndexOutOfBoundsException if the index is out of range
31       *         (<tt>index &lt; 0 || index &gt;= size()</tt>)
32       */
33      public abstract void setBoolean(int index, boolean value);
34      
35      /**
36       * Returns the number of elements in the collection.
37       * 
38       * @return the number of elements in the collection
39       */
40      public abstract int size();
41  
42      @Override
43      public boolean equals(Object obj) {
44          if (obj == this)
45              return true;
46          
47          if (obj instanceof ListBoolean) {
48              ListBoolean other = (ListBoolean) obj;
49  
50              if (size() != other.size())
51                  return false;
52  
53              for (int i = 0; i < size(); i++) {
54                  if (getBoolean(i) != other.getBoolean(i))
55                      return false;
56              }
57  
58              return true;
59          }
60          
61          return false;
62      }
63  
64      @Override
65      public int hashCode() {
66          int result = 1;
67          for (int i = 0; i < size(); i++) {
68              result = 31 * result + (getBoolean(i) ? 1 : 0);
69          }
70          return result;
71      }
72      
73  }
