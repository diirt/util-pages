5   package org.epics.util.array;
6   
7   import java.util.NoSuchElementException;
8   
9   /**
10   * An iterator for a stream of primitive numbers, which allows to retrieve
11   * the value casted in the type you prefer. This class allows to implement
12   * a single binding for iterating over a collection instead of six different
13   * binding. If the original type is required, instanceof can be used to
14   * differentiate between {@link IteratorByte}, {@link IteratorShort}, 
15   * {@link IteratorInt}, {@link IteratorLong}, {@link IteratorFloat} and 
16   * {@link IteratorDouble}.
17   * <p>
18   * We looked into making this class implement Iterator, but unfortunately,
19   * because of generics being invariant, we cannot provide a scheme that would work naturally
20   * in all cases. Ideally, we would want to have {@code IteratorNumber} be {@code Iterator<Number>}
21   * and {@code IteratorDouble} be {@code Iterator<Double>}, but this does
22   * not work because generics are invariant. We could have
23   * {@code Iterator<T extends Number>} and {@code Iterator<Double>}, but that would
24   * mean {@code IteratorNumber} would need a type parameter, and the user of the
25   * API would have to use bound type parameters like {@code IteratorNumber<? extends Number>},
26   * which is awful. We could have all extend {@code Iterator<Number>} but then
27   * the foreach loops would return {@code Number} in all cases, even for collections
28   * of more specific type.
29   *
30   * @author Gabriele Carcassi
31   */
32  public interface IteratorNumber {
33      
34      /**
35       * Returns {@code true} if the iteration has more elements.
36       * (In other words, returns {@code true} if {@code nextXxx} would
37       * return an element rather than throwing an exception.)
38       *
39       * @return {@code true} if the iteration has more elements
40       */
41      boolean hasNext();
42      
43      /**
44       * Returns the next element in the iteration casted to a float.
45       *
46       * @return the next element in the iteration
47       * @throws NoSuchElementException if the iteration has no more elements
48       */
49      float nextFloat();
50      
51      /**
52       * Returns the next element in the iteration casted to a double.
53       *
54       * @return the next element in the iteration
55       * @throws NoSuchElementException if the iteration has no more elements
56       */
57      double nextDouble();
58      
59      /**
60       * Returns the next element in the iteration casted to a byte.
61       *
62       * @return the next element in the iteration
63       * @throws NoSuchElementException if the iteration has no more elements
64       */
65      byte nextByte();
66      
67      /**
68       * Returns the next element in the iteration casted to a short.
69       *
70       * @return the next element in the iteration
71       * @throws NoSuchElementException if the iteration has no more elements
72       */
73      short nextShort();
74      
75      /**
76       * Returns the next element in the iteration casted to an int.
77       *
78       * @return the next element in the iteration
79       * @throws NoSuchElementException if the iteration has no more elements
80       */
81      int nextInt();
82  
83      /**
84       * Returns the next element in the iteration casted to a long.
85       *
86       * @return the next element in the iteration
87       * @throws NoSuchElementException if the iteration has no more elements
88       */
89      long nextLong();
90  }
