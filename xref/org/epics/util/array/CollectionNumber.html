5   package org.epics.util.array;
6   
7   import java.util.Collection;
8   
9   /**
10   * A collection of numeric (primitive) elements. It provides a size and
11   * can be iterated more than once.
12   * <p>
13   * The method names are taken from {@link Collection}, though not all
14   * methods are specified. At this moment, the class is read-only. If in the
15   * future the class is extended, the new methods should match the names from
16   * {@link Collection}.
17   *
18   * @author Gabriele Carcassi
19   */
20  public interface CollectionNumber {
21      
22      /**
23       * Returns an iterator over the elements of the collection.
24       * 
25       * @return a new iterator
26       */
27      IteratorNumber iterator();
28      
29      /**
30       * Returns the number of elements in the collection.
31       * 
32       * @return the number of elements in the collection
33       */
34      int size();
35  }
