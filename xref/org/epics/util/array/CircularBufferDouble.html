5   package org.epics.util.array;
6   
7   /**
8    * An implementation of a list on top of a circular buffer. The buffer
9    * will start at the initial capacity (default 10) and will continue
10   * to grow until the max capacity is reached. At that point, it will
11   * start to replace the oldest value with a newer one.
12   *
13   * @author carcassi
14   */
15  public class CircularBufferDouble extends ListDouble {
16      
17      private double[] data;
18      private int startOffset;
19      private int endOffset;
20      private final int maxCapacity;
21      private boolean reachedMax;
22  
23      /**
24       * Creates a new circular buffer with the given maximum capacity.
25       * 
26       * @param maxCapacity maximum capacity
27       */
28      public CircularBufferDouble(int maxCapacity) {
29          this(Math.min(10, maxCapacity), maxCapacity);
30      }
31  
32      /**
33       * Creates a new circular buffer with the given initial and maximum
34       * capacity.
35       * 
36       * @param initialCapacity initial capacity
37       * @param maxCapacity  maximum capacity
38       */
39      public CircularBufferDouble(int initialCapacity, int maxCapacity) {
40          data = new double[initialCapacity];
41          this.maxCapacity = maxCapacity;
42      }
43      
44      private void resize() {
45          int oldSize = data.length;
46          int newSize = oldSize * 2;
47          if (newSize > maxCapacity) {
48              newSize = maxCapacity + 1;
49              reachedMax = true;
50          }
51          double[] newData = new double[newSize];
52          System.arraycopy(data, 0, newData, 0, oldSize);
53          data = newData;
54      }
55  
56      /**
57       * {@inheritDoc }
58       */
59      @Override
60      public double getDouble(int index) {
61          index += startOffset;
62          if (index >= data.length) {
63              index -= data.length;
64          }
65          return data[index];
66      }
67  
68      /**
69       * {@inheritDoc }
70       */
71      @Override
72      public int size() {
73          int size = endOffset - startOffset;
74          if (size < 0) {
75              size += data.length;
76          }
77          return size;
78      }
79  
80      /**
81       * Adds a new value.
82       * 
83       * @param value new value
84       */
85      public void addDouble(double value) {
86          data[endOffset] = value;
87          endOffset++;
88          
89          // Grow the buffer if needed
90          if (endOffset == data.length && !reachedMax)
91              resize();
92          
93          // Loop over and advance the start point if needed
94          if (endOffset == data.length) {
95              endOffset = 0;
96          }
97          if (endOffset == startOffset)
98              startOffset++;
99          if (startOffset == data.length)
100             startOffset = 0;
101     }
102     
103     /**
104      * Removes all values from the buffer.
105      */
106     public void clear() {
107         startOffset = 0;
108         endOffset = 0;
109     }
110 
111     /**
112      * The maximum capacity for this circular buffer.
113      * 
114      * @return maximum capacity
115      */
116     public int getCurrentCapacity() {
117         return reachedMax ? maxCapacity : data.length;
118     }
119 }
