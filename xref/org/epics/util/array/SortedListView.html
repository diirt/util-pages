5   package org.epics.util.array;
6   
7   /**
8    * A sorted view of a list.
9    *
10   * @author carcassi
11   */
12  public class SortedListView extends ListDouble {
13  
14      private ListNumber values;
15      private ListInt indexes;
16      private int[] indexArray;
17  
18      SortedListView(ListNumber values) {
19          this.values = values;
20          indexArray = new int[values.size()];
21          for (int i = 0; i < indexArray.length; i++) {
22              indexArray[i] = i;
23          }
24          this.indexes = new ArrayInt(indexArray);
25      }
26  
27      SortedListView(ListNumber values, ListInt indexes) {
28          this.values = values;
29          this.indexes = indexes;
30          this.indexArray = null;
31      }
32  
33      @Override
34      public double getDouble(int index) {
35          return values.getDouble(indexes.getInt(index));
36      }
37  
38      @Override
39      public int size() {
40          return values.size();
41      }
42  
43      void exhange(int i, int j) {
44          int b = indexArray[i];
45          indexArray[i] = indexArray[j];
46          indexArray[j] = b;
47      }
48  
49      /**
50       * Returns the index map of the sorted view.
51       * 
52       * @return a list of integers
53       */
54      public ListInt getIndexes() {
55          return indexes;
56      }
57  
58      /**
59       * Quick-sort the view. The original list is left alone,
60       * and the internal list is modified.
61       * 
62       * @param list the view to sort
63       */
64      static void quicksort(SortedListView list) {
65          quicksort(list, 0, list.size() - 1);
66      }
67  
68      private static void quicksort(SortedListView list, int left, int right) {
69          if (right <= left) {
70              return;
71          }
72          int i = partition(list, left, right);
73          quicksort(list, left, i - 1);
74          quicksort(list, i + 1, right);
75      }
76  
77  // partition a[left] to a[right], assumes left < right
78      private static int partition(SortedListView list,
79              int left, int right) {
80          int i = left - 1;
81          int j = right;
82          while (true) {
83              while (less(list.getDouble(++i), list.getDouble(right)))      // find item on left to swap
84              ;                               // a[right] acts as sentinel
85              while (less(list.getDouble(right), list.getDouble(--j))) // find item on right to swap
86              {
87                  if (j == left) {
88                      break;           // don't go out-of-bounds
89                  }
90              }
91              if (i >= j) {
92                  break;                  // check if pointers cross
93              }
94              list.exhange(i, j);               // swap two elements into place
95          }
96          list.exhange(i, right);               // swap with partition element
97          return i;
98      }
99  
100     // is x < y ?
101     private static boolean less(double x, double y) {
102         return (x < y);
103     }
104 
105 }
