5   package org.epics.util.stats;
6   
7   /**
8    * Utility classes to compute ranges.
9    *
10   * @author carcassi
11   */
12  public class Ranges {
13      
14      /**
15       * Range from given min and max.
16       * 
17       * @param minValue minimum value
18       * @param maxValue maximum value
19       * @return the range
20       */
21      public static Range range(final double minValue, final double maxValue) {
22          if (minValue > maxValue) {
23              throw new IllegalArgumentException("minValue should be less then or equal to maxValue (" + minValue+ ", " + maxValue + ")");
24          }
25          return new Range() {
26  
27              @Override
28              public Number getMinimum() {
29                  return minValue;
30              }
31  
32              @Override
33              public Number getMaximum() {
34                  return maxValue;
35              }
36  
37              @Override
38              public String toString() {
39                  return Ranges.toString(this);
40              }
41              
42          };
43      }
44      
45      /**
46       * Determines whether the subrange is contained in the range or not.
47       * 
48       * @param range a range
49       * @param subrange a possible subrange
50       * @return true if subrange is contained in range
51       */
52      public static boolean contains(Range range, Range subrange) {
53          return range.getMinimum().doubleValue() <= subrange.getMinimum().doubleValue()
54                  && range.getMaximum().doubleValue() >= subrange.getMaximum().doubleValue();
55          
56      }
57  
58      /**
59       * Determines the range that can contain both ranges. If one of the
60       * ranges in contained in the other, the bigger range is returned.
61       * 
62       * @param range1 a range
63       * @param range2 another range
64       * @return the bigger range
65       */
66      public static Range sum(Range range1, Range range2) {
67          if (range1.getMinimum().doubleValue() <= range2.getMinimum().doubleValue()) {
68              if (range1.getMaximum().doubleValue() >= range2.getMaximum().doubleValue()) {
69                  return range1;
70              } else {
71                  return range(range1.getMinimum().doubleValue(), range2.getMaximum().doubleValue());
72              }
73          } else {
74              if (range1.getMaximum().doubleValue() >= range2.getMaximum().doubleValue()) {
75                  return range(range2.getMinimum().doubleValue(), range1.getMaximum().doubleValue());
76              } else {
77                  return range2;
78              }
79          }
80          
81      }
82  
83      /**
84       * Returns a String representation of the give range
85       * 
86       * @param range a range
87       * @return the string representation
88       */
89      public static String toString(Range range) {
90          return "[" + range.getMinimum() + " - " + range.getMaximum() + "]";
91      }
92      
93      /**
94       * Returns the value normalized within the range. It performs a linear
95       * transformation where the minimum value of the range becomes 0 while
96       * the maximum becomes 1.
97       * 
98       * @param range a range
99       * @param value a value
100      * @return the value transformed based on the range
101      */
102     public static double normalize(Range range, double value) {
103         return normalize(value, range.getMinimum().doubleValue(), range.getMaximum().doubleValue());
104     }
105     
106     private static double normalize(double value, double min, double max) {
107         return (value - min) / (max - min);
108     }
109 
110     /**
111      * Determines whether the value is contained by the range or not.
112      * 
113      * @param range a range
114      * @param value a value
115      * @return true if the value is within the range
116      */
117     public static boolean contains(Range range, double value) {
118         return value >= range.getMinimum().doubleValue() && value <= range.getMaximum().doubleValue();
119     }
120 }
